    Creates a new chip app

    module.exports = chip;

function chip(name, root) {
  var app = new App(name);
  app.initApp(root);
  return app;
}

chip.App = App;
chip.EventEmitter = require('./events');
chip.Controller = require('./controller');
chip.Router = require('./router');
