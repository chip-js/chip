	expr = filters.shift()
	if filters.length
		strIndex = expr.match(quoteExpr)?.length or 0
		filters.forEach (filter) ->
			args = filter.split(argSeparator)
			strings.splice strIndex++, 0, "'" + args[0] + "'"
			strIndex += filter.match(quoteExpr)?.length or 0
			args[0] = "''"
			expr = "runFilter(#{expr},#{args.join(',')})"
